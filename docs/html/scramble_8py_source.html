32def scramble_image(image, seed, p, n):
33 np.random.seed(seed)
34 height, width, channels = image.shape
35 scrambled_image = np.zeros_like(image)
36
37 # Разбиваем изображение на блоки 8x8
38 for i in range(0, height, 8):
39 for j in range(0, width, 8):
40 for c in range(channels):
41 block = image[i:i + 8, j:j + 8, c]
42 # Применяем дискретное косинусное преобразование (DCT)
43 dct_block = dct(dct(block.T, norm='ortho').T, norm='ortho')
44
45 # Создаем матрицу Bk с элементами -1 и 1
46 Bk = np.random.choice([1, -1], size=(8, 8), p=[p, 1 - p])
47
48 # Модифицируем матрицу DCT
49 for x in range(8):
50 for y in range(8):
51 if x >= n or y >= n:
52 dct_block[x, y] *= Bk[x, y]
53
54 # Применяем обратное дискретное косинусное преобразование (IDCT)
55 idct_block = idct(idct(dct_block.T, norm='ortho').T, norm='ortho')
56
57 # Ограничиваем значения в пределах [0, 1]
58 idct_block[idct_block < 0] = 0
59 idct_block[idct_block > 1] = 1
60
61 scrambled_image[i:i + 8, j:j + 8, c] = idct_block
62
63 return scrambled_image
